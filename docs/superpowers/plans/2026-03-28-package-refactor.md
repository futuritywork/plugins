# @futurity/plugins Package Refactor — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Refactor the plugins repo from a fork-and-build template into a Bun workspace monorepo with two publishable packages: `@futurity/plugins` (the library) and `create-futurity-plugin` (the scaffolder).

**Architecture:** Move all source files into `packages/plugins/`, add a build step that emits ESM JS + declarations into `dist/`, expose `app.fetch` as the primary handler API while keeping `app.listen()` as a convenience. Create a minimal scaffolder in `packages/create-plugin/` that copies a template directory and substitutes the project name.

**Tech Stack:** Bun (runtime, build, test, workspaces), TypeScript, zod, @modelcontextprotocol/sdk, jose

- [ ] **Step 3: Implement app.fetch**

Modify `packages/plugins/src/app.ts`. The core change: extract the request-handling logic from `StreamableHttpServer` into `McpApp` as a `fetch` method, and have `listen()` use it.

Replace the `listen` method and add `fetch` to the `McpApp` class:

```ts
// Add these fields to McpApp class:
private _httpHandler?: (req: Request) => Promise<Response>;
private _wsTransport?: WebSocketTransport;

/**
 * Fetch handler — the primary API for handling MCP requests.
 * Use with Bun.serve(), Cloudflare Workers, or any Request/Response runtime.
 *
 * Note: WebSocket transport is not supported via fetch. Use listen() for WS.
 */
get fetch(): (req: Request) => Promise<Response> {
  if (!this._httpHandler) {
    this._httpHandler = this.buildHttpHandler();
  }
  return this._httpHandler;
}

private buildHttpHandler(): (req: Request) => Promise<Response> {
  const path = this.options.path ?? "/mcp";
  let wellKnown: Record<string, WellKnownEntry> | undefined;
  let chainedAuthRouter:
    | ReturnType<typeof createChainedAuthRouter>
    | undefined;

  if (this.options.oauth) {
    const oauth = OAuthOptionsSchema.parse(this.options.oauth);
    const authServer: Record<string, unknown> = {
      issuer: oauth.issuer,
      authorization_endpoint: oauth.authorizationEndpoint,
      token_endpoint: oauth.tokenEndpoint,
      scopes_supported: oauth.scopesSupported ?? [],
      response_types_supported: oauth.responseTypesSupported ?? ["code"],
      grant_types_supported: oauth.grantTypesSupported ?? [
        "authorization_code",
      ],
    };
    if (oauth.jwksUri) authServer.jwks_uri = oauth.jwksUri;
    if (oauth.tokenEndpointAuthMethodsSupported) {
      authServer.token_endpoint_auth_methods_supported =
        oauth.tokenEndpointAuthMethodsSupported;
    }
    if (oauth.registrationEndpoint)
      authServer.registration_endpoint = oauth.registrationEndpoint;
    if (oauth.userInfoEndpoint)
      authServer.userinfo_endpoint = oauth.userInfoEndpoint;

    wellKnown = { "oauth-authorization-server": { body: authServer } };
  }

  if (this.options.pluginManifest) {
    const { signingKey, ...manifestData } = this.options.pluginManifest;
    const manifest = PluginManifestSchemaCompat.parse(manifestData);
    const payload = JSON.stringify(manifest);
    const jws = signPayload(payload, signingKey);

    wellKnown ??= {};
    wellKnown["futurity/plugin"] = {
      body: manifest,
      headers: { "X-Futurity-Signature": jws },
    };
  }

  if (this.options.chainedAuth) {
    chainedAuthRouter = createChainedAuthRouter(this.options.chainedAuth);
  }

  const httpServer = new StreamableHttpServer({
    port: 0, // Not used when calling handleFetch directly
    path,
    wellKnown,
    auth: this.options.auth,
    middleware: this.middlewares,
    chainedAuthRouter,
    onSession: async (transport: StreamableHttpTransport) => {
      const server = this.createServer();
      this.sessions.set(transport.sessionId, server);
      await server.connect(transport);
    },
    onSessionClose: (sessionId: string) => {
      this.sessions.delete(sessionId);
    },
  });

  this.httpServer = httpServer;

  return async (req: Request): Promise<Response> => {
    // Run middleware pipeline
    const runMiddlewares = async (
      index: number,
      req: Request
    ): Promise<Response | undefined> => {
      if (this.middlewares.length > 0 && index < this.middlewares.length) {
        return this.middlewares[index]?.(req, (nextReq) =>
          runMiddlewares(index + 1, nextReq)
        );
      }
      return httpServer.handleFetch(req);
    };

    const response = await runMiddlewares(0, req);
    return response || new Response("Not Found", { status: 404 });
  };
}
```

async listen(port: number, transportType: "websocket" | "http" = "http") {
  if (transportType === "websocket") {
    const path = this.options.path ?? "/mcp";
    let wellKnown: Record<string, WellKnownEntry> | undefined;

    if (this.options.oauth) {
      const oauth = OAuthOptionsSchema.parse(this.options.oauth);
      const authServer: Record<string, unknown> = {
        issuer: oauth.issuer,
        authorization_endpoint: oauth.authorizationEndpoint,
        token_endpoint: oauth.tokenEndpoint,
        scopes_supported: oauth.scopesSupported ?? [],
        response_types_supported: oauth.responseTypesSupported ?? ["code"],
        grant_types_supported: oauth.grantTypesSupported ?? [
          "authorization_code",
        ],
      };
      if (oauth.jwksUri) authServer.jwks_uri = oauth.jwksUri;
      if (oauth.tokenEndpointAuthMethodsSupported) {
        authServer.token_endpoint_auth_methods_supported =
          oauth.tokenEndpointAuthMethodsSupported;
      }
      if (oauth.registrationEndpoint)
        authServer.registration_endpoint = oauth.registrationEndpoint;
      if (oauth.userInfoEndpoint)
        authServer.userinfo_endpoint = oauth.userInfoEndpoint;
      wellKnown = { "oauth-authorization-server": { body: authServer } };
    }

    if (this.options.pluginManifest) {
      const { signingKey, ...manifestData } = this.options.pluginManifest;
      const manifest = PluginManifestSchemaCompat.parse(manifestData);
      const payload = JSON.stringify(manifest);
      const jws = signPayload(payload, signingKey);
      wellKnown ??= {};
      wellKnown["futurity/plugin"] = {
        body: manifest,
        headers: { "X-Futurity-Signature": jws },
      };
    }

    console.warn(
      "⚠️  WebSocket transport requires Bun runtime. It will not work in other environments."
    );
    const server = this.createServer();
    const transport = new WebSocketTransport({
      port,
      path,
      wellKnown,
      auth: this.options.auth,
      middleware: this.middlewares,
    });
    this._wsTransport = transport;
    await server.connect(transport);
  } else {
    // HTTP: use the fetch handler with Bun.serve
    const handler = this.fetch;
    this._bunServer = Bun.serve({
      hostname: "0.0.0.0",
      port,
      fetch: handler,
    });
  }

  const path = this.options.path ?? "/mcp";
  console.log(
    `MCP server listening on ${transportType}://localhost:${port}${path}`
  );
  return this;
}
```

Also add `_bunServer` field and update `stop()`:

```ts
private _bunServer?: ReturnType<typeof Bun.serve>;

async stop(): Promise<void> {
  this._bunServer?.stop();
  this._bunServer = undefined;
  await this.httpServer?.stop();
  this._wsTransport?.close();
  this.sessions.clear();
}
```

Remove the old `StreamableHttpServer.start()` call path from `listen()` — the `Bun.serve` in `listen()` now calls `this.fetch` directly, which creates the `StreamableHttpServer` internally via `buildHttpHandler()`.

However, `StreamableHttpServer` still needs its internal `handleFetch` exposed (no longer needs `start()` for the fetch path). The `start()` method on `StreamableHttpServer` is no longer called by `listen()` — instead, `listen()` creates a `Bun.serve` that calls `httpServer.handleFetch` via the middleware pipeline.

#!/usr/bin/env bun
import { existsSync, mkdirSync, cpSync, readFileSync, writeFileSync } from "node:fs";
import { resolve, basename } from "node:path";
import { execSync } from "node:child_process";

const arg = process.argv[2];

if (!arg) {
  console.error("Usage: bun create futurity-plugin <project-name>");
  console.error("       bun create futurity-plugin .  (scaffold in current directory)");
  process.exit(1);
}

const isCwd = arg === ".";
const targetDir = isCwd ? process.cwd() : resolve(process.cwd(), arg);
const projectName = isCwd ? basename(process.cwd()) : arg;

if (!isCwd) {
  if (existsSync(targetDir)) {
    console.error(`Error: Directory "${arg}" already exists.`);
    process.exit(1);
  }
  mkdirSync(targetDir, { recursive: true });
}

// Copy template files
const templateDir = resolve(import.meta.dirname, "template");
cpSync(templateDir, targetDir, { recursive: true });

// Process package.json template
const pkgPath = resolve(targetDir, "package.json.tmpl");
const pkgContent = readFileSync(pkgPath, "utf-8");
writeFileSync(
  resolve(targetDir, "package.json"),
  pkgContent.replace(/\{\{name\}\}/g, projectName)
);
// Remove the template file
import("node:fs").then((fs) => fs.unlinkSync(pkgPath));

// Install dependencies
console.log(`\nScaffolding ${projectName}...\n`);
execSync("bun install", { cwd: targetDir, stdio: "inherit" });

console.log(`\n✅ Created ${projectName}\n`);
console.log("Next steps:\n");
if (!isCwd) {
  console.log(`  cd ${arg}`);
}
console.log("  bun run src/index.ts\n");
```

## Quick Start

### Create a new plugin

```bash
bun create futurity-plugin my-plugin

bun run src/index.ts
```

### Or add to an existing project

```bash
bun add @futurity/plugins
```

### Basic usage

```ts
import { mcp, t } from "@futurity/plugins";

const app = mcp({
  name: "my-plugin",
  version: "1.0.0",
});

app.tool("hello", {
  description: "Say hello",
  input: t.obj({ name: t.str }),
  handler: async ({ name }) => ({
    greeting: `Hello, ${name}!`,
  }),
});

// Option A: Use the fetch handler (works in any runtime)
Bun.serve({ port: 3000, fetch: app.fetch });

// Option B: Convenience method (Bun only)
await app.listen(3000);
```
```
